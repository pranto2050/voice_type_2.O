Add a complete "Long Press to Listen" (Push-to-Talk) feature to the existing VoiceType Pro Electron desktop app. This must be fully working, beautifully animated, and configurable from settings. Every file change must be complete with no placeholders.

════════════════════════════════════════
 FEATURE OVERVIEW
════════════════════════════════════════

Name    : Long Press to Listen (Push-to-Talk mode)
Trigger : User holds a customizable key/button for
          3 seconds → voice recognition activates
          automatically. Release the key → recognition
          stops immediately and automatically.
Display : A beautiful full overlay animation appears
          on screen while the key is held, showing the
          app is actively listening.
Control : Can be fully enabled/disabled and the key
          can be customized from Settings → Shortcuts.

════════════════════════════════════════
 DETAILED BEHAVIOR SPEC
════════════════════════════════════════

STEP 1 — KEY HOLD DETECTION (3-second threshold):
  - User presses and HOLDS the configured key
    (default: Right Alt / Option key)
  - A circular progress ring starts filling up
    around the mic icon (0% → 100% over 3 seconds)
  - If user releases BEFORE 3 seconds:
    → ring resets, nothing happens, no listening
  - If user holds for 3 full seconds:
    → listening activates automatically
    → full animation kicks in (see below)
    → voice recognition starts via Python engine

STEP 2 — ACTIVE LISTENING STATE (key held):
  - As long as the key is held → recognition runs
  - Transcribed text is injected into focused input
  - Full overlay animation plays continuously
  - Status pill shows "LISTENING..." in green

STEP 3 — KEY RELEASE (auto-stop):
  - User releases the key → recognition stops instantly
  - Overlay animation fades out smoothly (300ms)
  - Progress ring resets to 0%
  - Status pill returns to idle state
  - A subtle "done" ripple effect plays once

/* ── Long Press to Listen ───────────────── */
let longPressTimer   = null;
let longPressActive  = false;
let longPressDuration = 3000; // ms — configurable
let holdKey = 'AltRight';     // default key — configurable
let holdFeatureEnabled = true; // toggle from settings

const CHARGE_DURATION = 3000;
let chargeStart = null;
let chargeAnimFrame = null;

// SVG ring circumference for r=52: 2 * Math.PI * 52 ≈ 326.7
const RING_CIRCUMFERENCE = 326.7;

function onKeyDown(e) {
  if (!holdFeatureEnabled) return;
  if (e.code !== holdKey)  return;
  if (longPressActive)     return;
  if (e.repeat)            return; // ignore key repeat

  chargeStart = Date.now();
  startChargeAnimation();

  longPressTimer = setTimeout(() => {
    activateLongPressListening();
  }, CHARGE_DURATION);
}

function onKeyUp(e) {
  if (e.code !== holdKey) return;
  clearTimeout(longPressTimer);
  cancelAnimationFrame(chargeAnimFrame);
  resetChargeRing();

  if (longPressActive) {
    deactivateLongPressListening();
  }
}

function startChargeAnimation() {
  function animate() {
    const elapsed  = Date.now() - chargeStart;
    const progress = Math.min(elapsed / CHARGE_DURATION, 1);
    const offset   = RING_CIRCUMFERENCE * (1 - progress);
    document.getElementById('charge-ring').style
      .strokeDashoffset = offset;

    // Show "Hold to activate..." label after 0.5s
    if (elapsed > 500) {
      document.getElementById('hold-label').style.opacity = 1;
    }

    if (progress < 1) {
      chargeAnimFrame = requestAnimationFrame(animate);
    }
  }
  chargeAnimFrame = requestAnimationFrame(animate);
}

function resetChargeRing() {
  cancelAnimationFrame(chargeAnimFrame);
  document.getElementById('charge-ring').style
    .strokeDashoffset = RING_CIRCUMFERENCE;
  document.getElementById('hold-label').style.opacity = 0;
}

function activateLongPressListening() {
  longPressActive = true;

  // 1. Play burst animation
  playActivationBurst();

  // 2. Start voice recognition
  window.electronAPI.startVoice();

  // 3. Show all active animations
  document.getElementById('listening-overlay').classList.add('active');
  document.getElementById('soundwave-bars').classList.add('active');
  document.getElementById('mic-float-bg').classList.add('active');
  document.getElementById('particle-orbit').classList.add('active');
  document.getElementById('status-pill').classList.add('listening');

  // 4. Update label
  document.getElementById('listening-label').classList.add('visible');
}

function deactivateLongPressListening() {
  longPressActive = false;

  // 1. Stop voice recognition
  window.electronAPI.stopVoice();

  // 2. Play release animation
  playReleaseAnimation();

  // 3. Remove all active classes after fade
  setTimeout(() => {
    document.getElementById('listening-overlay').classList.remove('active');
    document.getElementById('soundwave-bars').classList.remove('active');
    document.getElementById('mic-float-bg').classList.remove('active');
    document.getElementById('particle-orbit').classList.remove('active');
    document.getElementById('status-pill').classList.remove('listening');
    document.getElementById('listening-label').classList.remove('visible');
  }, 400);
}

function playActivationBurst() {
  const burst = document.getElementById('burst-container');
  burst.innerHTML = '';
  [0, 100, 200].forEach(delay => {
    const ring = document.createElement('div');
    ring.className = 'burst-ring';
    ring.style.animationDelay = delay + 'ms';
    burst.appendChild(ring);
  });
  setTimeout(() => burst.innerHTML = '', 900);
}

function playReleaseAnimation() {
  const ripple = document.createElement('div');
  ripple.className = 'release-ripple';
  document.getElementById('mic-container').appendChild(ripple);
  setTimeout(() => ripple.remove(), 600);
}

// Attach listeners
document.addEventListener('keydown', onKeyDown);
document.addEventListener('keyup',   onKeyUp);

// Also support mousedown/mouseup on HOLD button
const holdBtn = document.getElementById('hold-btn');
holdBtn.addEventListener('mousedown', () => {
  onKeyDown({ code: holdKey, repeat: false });
});
holdBtn.addEventListener('mouseup', () => {
  onKeyUp({ code: holdKey });
});
holdBtn.addEventListener('mouseleave', () => {
  onKeyUp({ code: holdKey });
});

const lpToggle = document.getElementById('lp-toggle');
const lpKeyField = document.getElementById('lp-key-field');
const lpDelaySlider = document.getElementById('lp-delay-slider');
const lpDelayLabel  = document.getElementById('lp-delay-label');

// Load saved values
const savedLP = JSON.parse(
  localStorage.getItem('longPress') || '{}');
const lpEnabled  = savedLP.enabled  ?? true;
const lpKey      = savedLP.key      ?? 'AltRight';
const lpKeyLabel = savedLP.keyLabel ?? 'Right Alt';
const lpDelay    = savedLP.delay    ?? 3000;

lpToggle.checked          = lpEnabled;
lpKeyField.textContent    = lpKeyLabel;
lpDelaySlider.value       = lpDelay / 1000;
lpDelayLabel.textContent  = (lpDelay/1000).toFixed(1) + 's';

// Sync to renderer
window.holdFeatureEnabled = lpEnabled;
window.holdKey            = lpKey;
window.longPressDuration  = lpDelay;

// Events
lpToggle.addEventListener('change', (e) => {
  window.holdFeatureEnabled = e.target.checked;
  localStorage.setItem('longPress', JSON.stringify({
    ...savedLP, enabled: e.target.checked }));
  document.getElementById('lp-options').style.display
    = e.target.checked ? 'block' : 'none';
});

lpDelaySlider.addEventListener('input', (e) => {
  const ms = Math.round(e.target.value * 1000);
  window.longPressDuration = ms;
  lpDelayLabel.textContent = e.target.value + 's';
  localStorage.setItem('longPress', JSON.stringify({
    ...JSON.parse(localStorage.getItem('longPress')||'{}'),
    delay: ms }));
});

let recordingKey = false;
lpKeyField.addEventListener('click', () => {
  recordingKey = true;
  lpKeyField.textContent = 'Press any key...';
  lpKeyField.classList.add('recording');
});
document.addEventListener('keydown', (e) => {
  if (!recordingKey) return;
  e.preventDefault();
  recordingKey = false;
  window.holdKey = e.code;
  const label = e.key === ' ' ? 'Space' :
    e.code.replace('Key','').replace('Digit','') || e.key;
  lpKeyField.textContent = label;
  lpKeyField.classList.remove('recording');
  localStorage.setItem('longPress', JSON.stringify({
    ...JSON.parse(localStorage.getItem('longPress')||'{}'),
    key: e.code, keyLabel: label }));
});
